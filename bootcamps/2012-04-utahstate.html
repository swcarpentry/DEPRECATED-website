Count lines in files or in standard input

import sys

def counter(source):
    count = 0
    for line in source:
        count += 1
    return count

filenames = sys.argv[1:]
if len(filenames) == 0:
    print 'STDIN', counter(sys.stdin)
else:
    for filename in filenames:
        reader = open(filename, 'r')
        count = counter(reader)
        print filename, count
